create table if not exists public.tauros_clients (
  id bigint primary key,
  discord text not null,
  email text not null unique,
  password text not null,
  created_at text
);

create table if not exists public.tauros_orders (
  id bigint primary key,
  product text not null,
  quantity int not null,
  total text not null,
  client text,
  contact text,
  date text,
  status text
);

alter table public.tauros_clients enable row level security;
alter table public.tauros_orders enable row level security;

drop policy if exists "clientes insert public" on public.tauros_clients;
drop policy if exists "clientes select public" on public.tauros_clients;
drop policy if exists "pedidos insert public" on public.tauros_orders;
drop policy if exists "pedidos select public" on public.tauros_orders;
drop policy if exists "pedidos delete public" on public.tauros_orders;

create policy "clientes insert public"
on public.tauros_clients for insert
to anon
with check (true);

create policy "clientes select public"
on public.tauros_clients for select
to anon
using (true);

create policy "pedidos insert public"
on public.tauros_orders for insert
to anon
with check (true);

create policy "pedidos select public"
on public.tauros_orders for select
to anon
using (true);

create policy "pedidos delete public"
on public.tauros_orders for delete
to anon
using (true);
